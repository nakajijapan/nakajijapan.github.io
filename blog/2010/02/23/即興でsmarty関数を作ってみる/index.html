February 23, 2010 ( Smarty ) 
function smarty_function_my_assign($params, &$smarty)
{
    $str="";
    foreach ($params as $key => $value) {
        if (substr($key, 0, 5) === "value") {
            $str .= $value;
        }
    }

    $smarty->assign($params['var'], $str);

}
