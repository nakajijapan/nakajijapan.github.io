#!/bin/bash
#find . -type f -name "*.html" -exec sed 's/Shift_JIS/UTF-8/g' {} \;

for FILE in $*;
do
cp $FILE $FILE.bak;
#sed -e "s/Shift_JIS/UTF-8/g" $FILE.bak > $FILE;
sed -e "s/EUC-JP/UTF-8/g" $FILE.bak > $FILE;

echo "end "$FILE
done
